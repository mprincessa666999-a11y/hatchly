final routerProvider = Provider<GoRouter>((ref) {
  // 1. Следим за авторизацией в Firebase
  final authState = ref.watch(authStateProvider);

  // 2. Следим за данными профиля (имя, фото)
  // Это КРИТИЧНО: когда имя изменится, роутер пересчитает redirect
  final profile = ref.watch(profileProvider);

  return GoRouter(
    initialLocation: '/splash',
    // Включаем логи для отладки (в консоли будет видно, куда переходит роутер)
    debugLogDiagnostics: true,

    redirect: (context, state) {
      // Если Firebase еще грузится — ждем
      if (authState.isLoading) return null;

      final isLoggedIn = authState.value != null;
      final isNameEmpty = profile.name.trim().isEmpty;

      final isSplash = state.matchedLocation == '/splash';
      final isLogin = state.matchedLocation == '/login';
      final isProfile = state.matchedLocation == '/profile';

      // ЛОГИКА ПРОВЕРКИ:

      // Шаг 1: Если не залогинен — только на логин или сплеш
      if (!isLoggedIn) {
        return (isLogin || isSplash) ? null : '/login';
      }

      // Шаг 2: Если залогинен, проверяем заполнение профиля
      if (isLoggedIn) {
        // Если имени НЕТ и мы не на странице профиля — принудительно идем на /profile
        if (isNameEmpty && !isProfile) {
          return '/profile';
        }

        // Если имя ЕСТЬ, но мы всё еще на страницах входа или настройки профиля — идем на Главную
        if (!isNameEmpty && (isLogin || isSplash || isProfile)) {
          return '/';
        }
      }

      return null;
    },
    routes: [
      GoRoute(
        path: '/splash',
        name: 'splash',
        builder: (context, state) =>
            SplashScreen(onFinished: () => context.go('/')),
      ),
      GoRoute(
        path: '/login',
        name: 'login',
        builder: (context, state) => const LoginScreen(),
      ),
      // Основное меню приложения (Табы)
      ShellRoute(
        builder: (context, state, child) => AppScaffold(child: child),
        routes: [
          GoRoute(
            path: '/',
            name: 'home',
            builder: (context, state) => const HomeScreen(),
          ),
          GoRoute(
            path: '/calendar',
            name: 'calendar',
            builder: (context, state) => const CalendarScreen(),
          ),
          GoRoute(
            path: '/notes',
            name: 'notes',
            builder: (context, state) => const NotesScreen(),
          ),
          GoRoute(
            path: '/partner',
            name: 'partner',
            builder: (context, state) => const PartnerScreen(),
          ),
        ],
      ),
      GoRoute(
        path: '/tasks/new',
        name: 'newTask',
        builder: (context, state) => const NewTaskScreen(),
      ),
      GoRoute(
        path: '/tasks/:id',
        name: 'taskDetail',
        builder: (context, state) =>
            TaskDetailScreen(taskId: state.pathParameters['id']!),
      ),
      // Экран настройки профиля вынесен из ShellRoute, чтобы не было BottomBar
      GoRoute(
        path: '/profile',
        name: 'profile',
        builder: (context, state) => const ProfileSetupScreen(),
      ),
      GoRoute(
        path: '/search',
        name: 'search',
        builder: (context, state) => const SearchScreen(),
      ),
    ],
  );
});

class SplashScreen extends StatefulWidget {

  const SplashScreen({super.key, required this.onFinished});

  @override
  State<SplashScreen> createState() => _SplashScreenState();
}

class _SplashScreenState extends State<SplashScreen>
    with SingleTickerProviderStateMixin {
  @override
  void initState() {
    super.initState();
    // Ждем 2 секунды и вызываем переход
    Future.delayed(const Duration(seconds: 2), () {
      if (mounted) widget.onFinished();
    });
  }

  @override
  Widget build(BuildContext context) {
    return const Scaffold(
      backgroundColor: Color(0xFF121212),
      body: Center(
        child: Text(
          'Couple App',
          style: TextStyle(
            color: Colors.white,
            fontSize: 32,
            fontWeight: FontWeight.bold,
          ),
        ),
      ),
    );
  }
}

class ProfileSetupScreen extends ConsumerStatefulWidget {
  const ProfileSetupScreen({super.key});

  @override
  ConsumerState<ProfileSetupScreen> createState() => _ProfileSetupScreenState();
}

class _ProfileSetupScreenState extends ConsumerState<ProfileSetupScreen> {
  late TextEditingController _nameController;

  @override
  void initState() {
    super.initState();
    // Читаем текущее имя из провайдера
    final profile = ref.read(profileProvider);
    _nameController = TextEditingController(text: profile.name);
  }

  @override
  void dispose() {
    _nameController.dispose();
    super.dispose();
  }

  void _saveName() async {
    final name = _nameController.text.trim();
    if (name.isEmpty) {
      ScaffoldMessenger.of(
        context,
      ).showSnackBar(const SnackBar(content: Text('Пожалуйста, введите имя')));
      return;
    }

    // 1. Сохраняем имя в провайдер
    ref.read(profileProvider.notifier).setName(name);

    // Убираем фокус с текстового поля (закрываем клавиатуру)
    FocusScope.of(context).unfocus();

    ScaffoldMessenger.of(context).showSnackBar(
      const SnackBar(
        content: Text('Профиль успешно настроен!'),
        duration: Duration(seconds: 1),
        backgroundColor: Colors.green,
      ),
    );

    // 2. ПЕРЕХОД
    // Ждем один кадр, чтобы состояние провайдера точно «просочилось» в приложение
    await Future.delayed(const Duration(milliseconds: 100));

    if (mounted) {
      // Используем go('/'), чтобы сбросить стек и зайти на главную
      context.go('/');
    }
  }

  void _generateCode() {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
    final rand = DateTime.now().millisecondsSinceEpoch;
    final code = List.generate(
      6,
      (i) => chars[(rand ~/ (i + 1)) % chars.length],
    ).join();
    ref.read(profileProvider.notifier).setInviteCode(code);
  }

  void _goBack() {
    if (context.canPop()) {
      context.pop();
    } else {
      context.go('/');
    }
  }

  @override
  Widget build(BuildContext context) {
    final profile = ref.watch(profileProvider);
    final isNameEmpty = _nameController.text.trim().isEmpty;

    return Scaffold(
      backgroundColor: AppColors.background,
      body: SafeArea(
        child: SingleChildScrollView(
          child: Column(
            children: [
              // Хедер
              Padding(
                padding: const EdgeInsets.fromLTRB(16, 12, 16, 0),
                child: Row(
                  children: [
                    IconButton(
                      onPressed: _goBack,
                      icon: const Icon(
                        Icons.arrow_back_ios_new_rounded,
                        color: AppColors.textPrimary,
                        size: 22,
                      ),
                    ),
                  ],
                ),
              ),
              const SizedBox(height: 32),

              // Аватар
              GestureDetector(
                onTap: () {
                  ScaffoldMessenger.of(context).showSnackBar(
                    const SnackBar(
                      content: Text(
                        'Выбор фото будет доступен в следующем обновлении',
                      ),
                    ),
                  );
                },
                child: Stack(
                  alignment: Alignment.center,
                  children: [
                    CircleAvatar(
                      radius: 55,
                      backgroundColor: AppColors.surface,
                      backgroundImage: profile.photoPath != null
                          ? AssetImage(profile.photoPath!)
                          : null,
                      child: profile.photoPath == null
                          ? const Icon(
                              Icons.person_outline,
                              color: AppColors.textSecondary,
                              size: 40,
                            )
                          : null,
                    ),
                    Positioned(
                      bottom: 0,
                      right: 0,
                      child: Container(
                        padding: const EdgeInsets.all(6),
                        decoration: const BoxDecoration(
                          color: AppColors.primary,
                          shape: BoxShape.circle,
                        ),
                        child: const Icon(
                          Icons.edit,
                          size: 14,
                          color: AppColors.white,
                        ),
                      ),
                    ),
                  ],
                ),
              ),

              const SizedBox(height: 20),

              // Поле ввода имени
              Padding(
                padding: const EdgeInsets.symmetric(horizontal: 32),
                child: TextField(
                  controller: _nameController,
                  textAlign: TextAlign.center,
                  style: AppTextStyles.h2.copyWith(
                    color: AppColors.textPrimary,
                  ),
                  decoration: InputDecoration(
                    hintText: 'Введите имя',
                    hintStyle: AppTextStyles.h2.copyWith(
                      color: AppColors.textSecondary,
                    ),
                    border: InputBorder.none,
                  ),
                  onChanged: (val) => setState(() {}),
                ),
              ),

              const SizedBox(height: 32),

              Padding(

import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:go_router/go_router.dart';
import 'package:couple_app/core/theme/app_colors.dart';
import 'package:couple_app/core/ui/app_icons.dart';

class AppScaffold extends ConsumerStatefulWidget {
  final Widget child;
  const AppScaffold({super.key, required this.child});

  @override
  ConsumerState<AppScaffold> createState() => _AppScaffoldState();
}

class _AppScaffoldState extends ConsumerState<AppScaffold> {
  // Список путей для табов
  final List<String> _tabs = ['/', '/calendar', '/notes', '/partner'];

  int _calculateSelectedIndex(BuildContext context) {
    final String location = GoRouterState.of(context).matchedLocation;

    // Если мы на главной, возвращаем 0
    if (location == '/') return 0;

    // Ищем точное совпадение или принадлежность к разделу (кроме корня)
    for (int i = 1; i < _tabs.length; i++) {
      if (location.startsWith(_tabs[i])) {
        return i;
      }
    }
    return 0;
  }

  void _onTabTapped(int index) {
    // Используем go, чтобы не плодить стек навигации внутри табов
    context.go(_tabs[index]);
  }

  @override
  Widget build(BuildContext context) {
    final int currentIndex = _calculateSelectedIndex(context);

    return Scaffold(
      backgroundColor: AppColors.background,
      // Используем Key, чтобы Flutter понимал, когда нужно перерисовать body
      body: widget.child,
      bottomNavigationBar: _BottomNav(
        currentIndex: currentIndex,
        onTap: _onTabTapped,
      ),
    );
  }
}

class _BottomNav extends StatelessWidget {
  final int currentIndex;
  final ValueChanged<int> onTap;
  const _BottomNav({required this.currentIndex, required this.onTap});

  @override
  Widget build(BuildContext context) {
    return Container(
      // Увеличили высоту для комфортного нажатия
      height: 75 + MediaQuery.of(context).padding.bottom,
      padding: EdgeInsets.only(bottom: MediaQuery.of(context).padding.bottom),
      decoration: const BoxDecoration(
        color: AppColors.background,
        border: Border(
          top: BorderSide(
            color: Color(0xFF2C2C2C),
            width: 0.5,
          ), // AppColors.border
        ),
      ),
      child: Row(
        mainAxisAlignment: MainAxisAlignment.spaceAround,
        children: [
          _NavItem(
            index: 0,
            currentIndex: currentIndex,
            onTap: onTap,
            icon: AppIcons.home,
          ),
          _NavItem(
            index: 1,
            currentIndex: currentIndex,
            onTap: onTap,
            icon: AppIcons.calendar,
          ),
          _NavItem(
            index: 2,
            currentIndex: currentIndex,
            onTap: onTap,
            icon: AppIcons.notes,
          ),
          _NavItem(
            index: 3,
            currentIndex: currentIndex,
            onTap: onTap,
            icon: AppIcons.lists, // Проверь, что в AppIcons это именно lists
          ),
        ],
      ),
    );
  }
}

class _NavItem extends StatelessWidget {
  final int index;
  final int currentIndex;
  final ValueChanged<int> onTap;
  final Widget Function({double size, Color? color}) icon;

  const _NavItem({
    required this.index,
    required this.currentIndex,
    required this.onTap,
    required this.icon,
  });

  bool get isActive => index == currentIndex;

  @override
  Widget build(BuildContext context) {
    // Цвет активной и неактивной иконки
    final color = isActive
        ? AppColors.primary
        : const Color(0xFF8E8E8E); // AppColors.iconInactive

    return GestureDetector(
      onTap: () => onTap(index),
      behavior: HitTestBehavior.opaque,
      child: SizedBox(
        width: 65,
        height: double.infinity,
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            icon(size: 26, color: color),
            const SizedBox(height: 4),
            // Небольшая точка под активным табом (опционально)
            AnimatedContainer(
              duration: const Duration(milliseconds: 200),
              width: isActive ? 4 : 0,
              height: 4,
              decoration: BoxDecoration(color: color, shape: BoxShape.circle),
            ),
          ],
        ),
      ),
    );
  }
}

import 'package:flutter_riverpod/flutter_riverpod.dart';

class UserProfile {
  final String name;
  final String? photoPath;
  final String? inviteCode;
  final String? partnerName;

  const UserProfile({
    this.name = '', // По умолчанию пусто
    this.photoPath,
    this.inviteCode,
    this.partnerName,
  });

  // Вспомогательный метод, чтобы в роутере было удобно проверять
  bool get isNotSet => name.trim().isEmpty;

  UserProfile copyWith({
    String? name,
    String? photoPath,
    String? inviteCode,
    String? partnerName,
  }) {
    return UserProfile(
      name: name ?? this.name,
      photoPath: photoPath ?? this.photoPath,
      inviteCode: inviteCode ?? this.inviteCode,
      partnerName: partnerName ?? this.partnerName,
    );
  }
}

class ProfileNotifier extends StateNotifier<UserProfile> {
  // ИСПРАВЛЕНО: Убрали "Кузя Лакомкин" по умолчанию.
  // Теперь при первом запуске имя пустое, и приложение поймет, что нужно его спросить.
  ProfileNotifier() : super(const UserProfile(name: ''));

  void setName(String name) => state = state.copyWith(name: name);
  void setPhoto(String path) => state = state.copyWith(photoPath: path);
  void setInviteCode(String code) => state = state.copyWith(inviteCode: code);
  void setPartnerName(String name) => state = state.copyWith(partnerName: name);
}

final profileProvider = StateNotifierProvider<ProfileNotifier, UserProfile>((
  ref,
) {
  return ProfileNotifier();
});
